206 void static keyCallback(GLFWwindow *window, int key, int scancode, int action, int modifiers) {
207 auto keyboard = (IEKeyboard *) ((IEWindowUser *) (glfwGetWindowUserPointer(window)))->IEKeyboard; // keyboard connected to the window
208 if (action == GLFW_REPEAT) {
209 keyboard->queue.emplace_back(key, action);
210 return;
211 }
212 IEKeyPressDescription thisKeyPress{key, action};
213 IEKeyPressDescription oppositeKeyPress{key, thisKeyPress.action == GLFW_PRESS ? GLFW_RELEASE : GLFW_PRESS};
214 auto oppositeKeyPressIterator = std::find(keyboard->queue.begin(), keyboard->queue.end(), oppositeKeyPress);
215 if (oppositeKeyPressIterator != keyboard->queue.end()) {
216 keyboard->queue.erase(oppositeKeyPressIterator);
217 }
218 keyboard->queue.push_back(thisKeyPress);
219 }
