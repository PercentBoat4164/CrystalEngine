201 void static keyCallback(GLFWwindow* window, int key, int scancode, int action, int modifiers) {
202 if (action == GLFW_REPEAT) {
203 return;
204 }
205 auto keyboard = (IEKeyboard *)((IEWindowUserPointer *)(glfwGetWindowUserPointer(window)))->keyboard; // keyboard connected to the window
206 IeKeyPressDescription thisKeyPress{key, action};
207 IeKeyPressDescription oppositeKeyPress{key, thisKeyPress.action == GLFW_PRESS ? GLFW_RELEASE : GLFW_PRESS};
208 auto oppositeKeyPressIterator = std::find(keyboard->queue.begin(), keyboard->queue.end(), oppositeKeyPress);
209 if (oppositeKeyPressIterator != keyboard->queue.end()) {
210 keyboard->queue.erase(oppositeKeyPressIterator);

212 keyboard->queue.push_back(thisKeyPress);
