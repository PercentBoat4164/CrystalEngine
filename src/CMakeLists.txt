add_dependencies(IlluminationEngine IEGraphicsModule)

addVulkan(IlluminationEngine)

    execute_process(COMMAND rmdir ${CMAKE_BINARY_DIR}/VulkanFOUND)

endif ()
if (EXISTS ${CMAKE_BINARY_DIR}/OpenGLFOUND)
    execute_process(COMMAND rmdir ${CMAKE_BINARY_DIR}/OpenGLFOUND)
    add_compile_definitions(ILLUMINATION_ENGINE_OPENGL)
    addGLFW(IlluminationEngine)
endif ()
